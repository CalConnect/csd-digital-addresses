=== IxAddressProfile

[source,json]
----
{

  "id": "http://www.iso.org/tc211/tc211-sample.adp",
  "type": "iso-19160-address-profile",
  "publisher": "http://www.iso.org/tc211/",
  "signature": "...",

  "name": "TC 211 Minimal Address Profile",
  "localization": {
    "language": "en",
    "script": "en"
  },

  "area": {
    "countries": ["uk"]
  },

  "dataTypes": {
    "addressedObjectIdentifier": {
      "primitiveType": "Integer"
    }
  },
  "addressComponents": { ... },
  "addressClasses": { ... }
}
----
